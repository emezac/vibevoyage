def handler_description
  case @handler
  when Proc
    if @handler.lambda?
      "Lambda with #{@handler.arity} parameters"
    else
      "Proc with #{@handler.arity} parameters"
    end
  else
    "Callable object (#{@handler.class})"
  end
end
