def initialize(task_id:, name:, input_data: {}, is_llm_task: false, tool_name: nil, max_retries: 0)
  @task_id = task_id
  @name = name
  @status = :pending
  @input_data = input_data
  @output_data = {}
  @is_llm_task = is_llm_task
  @tool_name = tool_name
  @max_retries = max_retries
  @retry_count = 0
end
